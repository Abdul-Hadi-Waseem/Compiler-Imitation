def extract_data(filename):
    """
    Args: filename [string] name of the sheesh file
    returns dataframe
    """
    try:
        data = pd.read_csv(filename, sep='\t')
        data = data.values
        return data
    except:
        # check if file exists
        if not os.path.exists(filename):
            print("ERR: File does not exists")
            FileNotFoundError()
            sys.exit(1)
